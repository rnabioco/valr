Merge overlapping intervals.

Usage

bed_merge(x, max_dist = 0, strand = FALSE, ...)

Arguments

x
tbl of intervals
max_dist
maximum distance between intervals to merge
strand
merge intervals on the same strand
...
name-value pairs that specify merging operations

Value

data_frame

Description

Merge overlapping intervals.

Examples

x <- tibble::tribble(
 ~chrom, ~start, ~end, ~value, ~strand,
 "chr1", 1,      50,   1,      '+',
 "chr1", 100,    200,  2,      '+',
 "chr1", 150,    250,  3,      '-',
 "chr2", 1,      25,   4,      '+',
 "chr2", 200,    400,  5,      '-',
 "chr2", 400,    500,  6,      '+',
 "chr2", 450,    550,  7,      '+'
)

bed_merge(x)


# A tibble: 4 × 3
  chrom start   end
  <chr> <dbl> <dbl>
1  chr1     1    50
2  chr1   100   250
3  chr2     1    25
4  chr2   200   550



bed_merge(x, max_dist = 100)


# A tibble: 3 × 3
  chrom start   end
  <chr> <dbl> <dbl>
1  chr1     1   250
2  chr2     1    25
3  chr2   200   550



bed_merge(x, strand = TRUE)


# A tibble: 6 × 4
  chrom strand start   end
  <chr>  <chr> <dbl> <dbl>
1  chr1      +     1    50
2  chr1      +   100   200
3  chr1      -   150   250
4  chr2      +     1    25
5  chr2      -   200   400
6  chr2      +   400   550



bed_merge(x, .value = sum(value))


# A tibble: 4 × 4
  chrom start   end .value
  <chr> <dbl> <dbl>  <dbl>
1  chr1     1    50      1
2  chr1   100   250      5
3  chr2     1    25      4
4  chr2   200   550     18
